import { Injectable, inject } from '@angular/core'; import { HttpClient } from '@angular/common/http'; @Injectable({ providedIn: 'root' }) export class AdminService { private apiUrl = 'http://localhost:3000/api/admin'; private http = inject(HttpClient); // Lekéri a dolgozókat a kiszámolt órákkal getStatisztikak(datum: string) { return this.http.get(`${this.apiUrl}/statisztikak?datum=${datum}`); } // Felhasználó végleges törlése felhasznaloTorlese(id: string) { return this.http.delete(`${this.apiUrl}/felhasznalo/${id}`); } uzenetKuldese(id: string, uzenet: string) { return this.http.put(`${this.apiUrl}/felhasznalo/${id}/uzenet`, { uzenet }); } }
